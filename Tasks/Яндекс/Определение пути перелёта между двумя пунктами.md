```js
function findPath(start, end, fetchFlights) {
  const queue = [{ city: start, path: [start] }];
  const visited = new Set([start]);

  async function search() {
    while (queue.length) {
      const { city, path } = queue.shift();

      if (city === end) return path;

      const flights = await fetchFlights(city);

      for (const next of flights) {
        if (!visited.has(next)) {
          visited.add(next);
          queue.push({ city: next, path: [...path, next] });
        }
      }
    }

    throw new Error('No way');
  }

  return search();
}
```
