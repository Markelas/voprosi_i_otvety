/** Сделать асинх. функцию, которая примет в виде промисов неограниченное кол-во аргументов и выведет их сумму promiseSum(Promise.resolve(1), Promise.resolve(3).then(cnosle.log)) // 4**/

async function promiseSum(...promises) {
  const values = await Promise.all(promises);
  return values.reduce((sum, num) => sum + num, 0);
}

// Использование:
promiseSum(Promise.resolve(1), Promise.resolve(3))
  .then(console.log); // 4
