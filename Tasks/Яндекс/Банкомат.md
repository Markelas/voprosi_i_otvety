С лимитами

```js
function withdrawMoney(amount, atmLimits) {
    const denominations = [5000, 1000, 500, 100];
    let result = {};

    if (amount % 100 !== 0) {
        return "Сумма должна быть кратна 100";
    }

    for (let nominal of denominations) {
        let count = Math.floor(amount / nominal);
        // Берем или необходимое кол-во, или доступное (лимит)
        let banknotesToTake = Math.min(count, atmLimits[nominal] || 0);

        if (banknotesToTake > 0) {
            result[nominal] = banknotesToTake;
            amount -= banknotesToTake * nominal;
            atmLimits[nominal] -= banknotesToTake; // Обновляем лимит в банкомате
        }
    }

    return amount === 0 ? result : "Недостаточно купюр для выдачи";
}

// Пример использования:
let limits = { 5000: 2, 1000: 5, 500: 10, 100: 20 };
console.log(withdrawMoney(7500, limits)); // { '5000': 1, '1000': 2, '500': 1 }
console.log(limits); // Обновленные лимиты
```
