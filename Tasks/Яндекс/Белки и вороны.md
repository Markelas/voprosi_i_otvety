* * которая находит всех белок на дереве и возвращает их имена

/*
Type Tree = { nest: Squirrel | Raven; branches?: Tree[]; }
Type Squirrel = { name: string; type: 'squirrel';}
Type Raven = { name: string; type: 'raven'; }
*/


function squirrelScanner(tree) { // Пушистик
    const { nest, branches } = tree;
    const result = [];

    if (nest && nest.type === 'squirrel') {
        result.push(nest.name)
    }

    if ( branches && branches.length) {
        for (const branch of branches) {
            result.push(...squirrelScanner(branch));
        }
    }
    
    return result;
}

function squirrelScanner(tree) { 
    const stack = [tree]; // [Черное Крыло, Пушистик]
    const result = []; // [Рыжик]

    while (stack?.length) {
        const currentNode = stack.pop(); // 
        const { branches = [] } = currentNode;

        for (let i = branches.length - 1; i >= 0; i--) {
            const branch = branches[i];
            stack.push(branch);
        }

        if (currentNode.nest.type === 'squirrel') {
            result.push(currentNode.nest.name)
        }
    }

    return result;
}
