```js
/*
Необходимо написать функцию compress, которая принимает массив чисел и возвращает строку,
представляющую собой сжатое представление диапазонов чисел.
Правила:
- Если числа идут подряд (с шагом 1), они объединяются в диапазон "начало-конец"
- Если число не входит в диапазон, оно записывается отдельно
- Числа в массиве могут быть в любом порядке (нужно отсортировать)
- Пример: [1,2,3,5,6,8,9,11,0] -> "0,1-3,5-6,8-9,11"
*/

function compress(list) {
    // Если массив пустой, возвращаем пустую строку
    if (list.length === 0) return '';
    
    // 1. СОРТИРУЕМ ЧИСЛА ПО ВОЗРАСТАНИЮ
    // Создаём копию массива через [...list], чтобы не мутировать исходный
    const sorted = [...list].sort((a, b) => a - b);
    // Пример: [1,2,3,5,6,8,9,11,0] -> [0,1,2,3,5,6,8,9,11]
    
    // Массив для хранения частей результата (например, ["0", "1-3", "5-6", "8-9", "11"])
    const result = [];
    
    // Начало текущего диапазона
    let start = sorted[0]; // start = 0
    
    // Предыдущее обработанное число
    let prev = sorted[0];   // prev = 0
    
    // 2. ПРОХОДИМ ПО ОТСОРТИРОВАННОМУ МАССИВУ
    // i <= sorted.length, чтобы обработать последний элемент
    for (let i = 1; i <= sorted.length; i++) {
        const current = sorted[i]; // текущее число
        
        // 3. ПРОВЕРЯЕМ, ЯВЛЯЕТСЯ ЛИ ТЕКУЩЕЕ ЧИСЛО ПРОДОЛЖЕНИЕМ ДИАПАЗОНА
        // Числа идут подряд, если текущее = предыдущее + 1
        if (current === prev + 1) {
            // Продолжаем диапазон
            prev = current; // обновляем предыдущее число
        } else {
            // 4. ДИАПАЗОН ЗАКОНЧИЛСЯ - СОХРАНЯЕМ ЕГО В РЕЗУЛЬТАТ
            
            // Если в диапазоне только одно число (start === prev)
            if (start === prev) {
                result.push(start.toString()); // добавляем просто число
            } else {
                // Если в диапазоне несколько чисел, добавляем "начало-конец"
                result.push(`${start}-${prev}`);
            }
            
            // 5. НАЧИНАЕМ НОВЫЙ ДИАПАЗОН
            start = current; // новое начало диапазона
            prev = current;  // новый предыдущий элемент
        }
    }
    
    // 6. ОБЪЕДИНЯЕМ ВСЕ ЧАСТИ ЧЕРЕЗ ЗАПЯТУЮ
    return result.join(',');
}
```
