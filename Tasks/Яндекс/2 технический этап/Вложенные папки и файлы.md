```js
function printDirectoryStructure(root) {
    // Проверяем, что входные данные корректны
    if (!root || typeof root !== 'object') return;

    // Начинаем с корневого элемента на глубине 0
    const stack = [{ node: root, depth: 0 }];

    // Пока стек не пуст - обрабатываем элементы
    while (stack.length) {
        // Извлекаем последний добавленный элемент
        const { node, depth } = stack.pop();
        // Например: node = { name: "src", children: [...] }, depth = 0

        // Получаем имя текущего узла и его дочерние элементы
        const { name, children } = node;
        // name = "src", children = [...]

        console.log(`${'  '.repeat(depth)}${name}`);

        if (Array.isArray(children)) {
            // Проходим по children с конца, чтобы сохранить порядок
            // при обходе в глубину (сначала обработаем последний добавленный)
            for (let i = children.length - 1; i >= 0; i--) {
                // Добавляем каждый дочерний элемент в стек с увеличенной глубиной
                stack.push({ node: children[i], depth: depth + 1 });
                // stack = [ { node: child3, depth: 1 }, { node: child2, depth: 1 }, { node: child1, depth: 1 } ]
            }
        }
    }
}
```

С рекурсией
```js
function printDirectoryStructureRecursive(root, depth = 0) {
    // Базовый случай: проверяем корректность данных
    if (!root || typeof root !== 'object') return;

    // Выводим текущий узел
    const { name, children } = root;
    console.log(`${'  '.repeat(depth)}${name}`);

    // Если есть дочерние элементы - рекурсивно обрабатываем их
    if (Array.isArray(children)) {
        for (const child of children) {
            // Рекурсивный вызов с увеличением глубины
            printDirectoryStructureRecursive(child, depth + 1);
        }
    }
}
```
