// function parallelLimit(urls, limit, callback)
// Нужно реализовать функцию parallelLimit, которая опрашивает
// массив URL-ов в заданном лимите параллельных запросов.
// Все запросы должны выполняться в порядке следования URL в
// массиве.
// Требования:
// * Лимит limit — максимум одновременных запросов
// * После завершения любого запроса — сразу отправляется
// следующий
// * Результаты должны сохраняться в порядке оригинального
// массива URL-ов
// * Один и тот же URL не должен запрашиваться дважды —
// должна быть мемоизация
// * Ошибки обрабатывать не нужно
// Используется fetch(url): Promise<string>

function parallelLimit(urls, limit, callback) {
  // Массив для результатов в правильном порядке
  const results = [];
  
  // Кеш для мемоизации: url -> Promise
  // Если один URL встретится дважды, используем сохраненный промис
  const cache = new Map();
  
  // Индекс следующего URL для запуска
  let nextIndex = 0;
  
  // Счетчик завершенных запросов
  let completed = 0;

  // Основная функция запуска запросов
  function run() {
    // Если все запросы завершены - вызываем callback с результатами
    if (completed === urls.length) {
      callback(results);
      return;
    }

    // Запускаем новые запросы, пока есть место (не больше limit активных)
    // nextIndex - completed = количество активных запросов
    while (nextIndex < urls.length && nextIndex - completed < limit) {
      const index = nextIndex++;  // берем следующий индекс и увеличиваем счетчик
      const url = urls[index];    // получаем URL

      // Мемоизация: если URL еще не запрашивали - создаем промис
      if (!cache.has(url)) {
        cache.set(url, fetch(url));
      }

      // Запускаем запрос и обрабатываем результат
      cache.get(url).then(data => {
        results[index] = data;    // сохраняем результат по правильному индексу
        completed++;              // увеличиваем счетчик завершенных
        run();                    // запускаем следующий запрос
      });
    }
  }

  // Стартуем процесс
  run();
}
