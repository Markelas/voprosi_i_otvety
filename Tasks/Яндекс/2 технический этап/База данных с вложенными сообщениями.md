function toTree(items) {
  // Создаём Map, где ключ = path, значение = узел с полями path, text и replies
  // Map даёт более чёткое представление о том, что мы делаем поиск по ключу
  const nodeMap = new Map();
  
  // Массив для корневых элементов (у которых нет родителя)
  const roots = [];
  
  // Шаг 1: Создаём узлы и помещаем их в Map
  // Проходим по всем сообщениям из входного массива
  items.forEach(item => {
    // Для каждого сообщения создаём узел:
    const node = {
      path: item.path,
      text: item.text,
      replies: [] // пока пусто, заполним на следующем шаге
    };
    
    // Сохраняем узел в Map, используя path как ключ
    // Теперь мы можем быстро найти любой узел по его path
    nodeMap.set(item.path, node);
  });
  
  // Шаг 2: Расставляем узлы по родителям
  // Снова проходим по всем исходным сообщениям
  items.forEach(item => {
    // Получаем узел из Map по его path
    // {path: "1.1.1", text: "Как дела?", replies: []}
    const currentNode = nodeMap.get(item.path);
    
    // Находим путь родителя, отрезая последнюю часть
    // Например: "1.2.3" -> "1.2"
    // Удаляем последний элемент
    const parentPath = item.path.substring(0, item.path.lastIndexOf('.'));
    
    // Проверяем, есть ли родительский путь и существует ли он в Map
    if (parentPath && nodeMap.has(parentPath)) {
      // Если родитель найден, добавляем текущий узел в его replies
      const parentNode = nodeMap.get(parentPath);
      parentNode.replies.push(currentNode);
    } else {
      // Если родителя нет (пустой parentPath или нет в Map),
      // значит это корневой элемент
      roots.push(currentNode);
    }
  });
  
  // Шаг 3: Удаляем пустые массивы replies для красоты
  // Рекурсивно проходим по всему дереву
  function removeEmptyReplies(node) {
    // Если у узла есть дочерние элементы
    if (node.replies.length > 0) {
      // Рекурсивно обрабатываем каждого ребёнка
      node.replies.forEach(child => removeEmptyReplies(child));
    } else {
      // Если дочерних элементов нет, удаляем свойство replies
      delete node.replies;
    }
  }
  
  // Применяем функцию ко всем корневым узлам
  roots.forEach(root => removeEmptyReplies(root));
  
  return roots;
}
