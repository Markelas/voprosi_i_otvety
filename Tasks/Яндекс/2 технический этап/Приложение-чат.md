```js
// 1 приложение-чат function solution(connect, render)
// приложение-чат должно отображать новые сообщения,
// которые приходят с сервера, как можно быстрее
// Сообщение имеет формат: interface Message {id:number; text:string}
// Id самого первого сообщения = 1, a id каждого следующего
// сообщения на 1 больше, чем id предыдущего.
// Нужно выводить сообщения в правильном порядке, однако
// сервер не гарантирует правильный порядок сообщений.
// Таймлайн:
// (приходит) 7 1 2 3 6 5 4  8
// (рисуем) . 1 2 3 .. 4 5 6 7 8
// Сообщения от сервера приходят в обработчик функции
connect: connect(function onMessage(message){...});
// Отображать сообщения нужно с помощью функции
render(message)
```

```js
function solution(connect, render) {
  // Хранилище для всех полученных сообщений
  // Ключ - id сообщения, значение - само сообщение
  const received = new Map();
  
  // Указатель на следующее сообщение, которое нужно отрисовать
  // Начинаем с 1, потому что самое первое сообщение имеет id = 1
  let nextToRender = 1;

  // Подключаемся к серверу и устанавливаем обработчик
  connect(function onMessage(message) {
    // 1. Сохраняем полученное сообщение
    received.set(message.id, message);

    // 2. Пытаемся отрисовать все возможные сообщения подряд
    // Проверяем, есть ли у нас сообщение с id = nextToRender
    while (received.has(nextToRender)) {
      // Если есть - отрисовываем его
      render(received.get(nextToRender));
      
      // Удаляем из хранилища (уже не нужно)
      received.delete(nextToRender);
      
      // Увеличиваем счётчик следующего ожидаемого сообщения
      nextToRender++;
    }
  });
}
```

```js
const wait = () => new Promise(resolve => setTimeout(resolve, 500));
const runTest = async () => {
    const renderedMessages = [];
    const expectedMessages = [
        { id: 1, text: 'One' },
        { id: 2, text: 'Two' },
        { id: 3, text: 'Three' },
        { id: 4, text: 'Four' },
        { id: 5, text: 'Five' },
        { id: 6, text: 'Six' },
        { id: 7, text: 'Seven' },
        { id: 8, text: 'Eight' }
    ];
    function testRender(message) {
        console.log('Rendered message: ', message);
        renderedMessages.push(message);
    }
    // Создаём мок для connect
    let messageHandler = null;
    
    function mockConnect(handler) {
        messageHandler = handler; // Сохраняем обработчик
    }
    // Вызываем solution с мок-connect
    solution(mockConnect, testRender);
    // Теперь эмулируем получение сообщений через сохранённый обработчик
    messageHandler({ id: 7, text: 'Seven' });
    await wait();
    messageHandler({ id: 1, text: 'One' });
    await wait();
    messageHandler({ id: 2, text: 'Two' });
    await wait();
    messageHandler({ id: 3, text: 'Three' });
    await wait();
    messageHandler({ id: 6, text: 'Six' });
    await wait();
    messageHandler({ id: 5, text: 'Five' });
    await wait();
    messageHandler({ id: 4, text: 'Four' });
    await wait();
    messageHandler({ id: 8, text: 'Eight' });
    // Ждём обработки последнего сообщения
    await wait(100);
    if (JSON.stringify(renderedMessages) !== JSON.stringify(expectedMessages)) {
        console.error(`Expected: ${JSON.stringify(expectedMessages)}`);
        console.error(`Received: ${JSON.stringify(renderedMessages)}`);
        return;
    }
    console.log('Test passed!');
    console.log('Order:', renderedMessages.map(m => m.id).join(' → '));
};
// Запускаем тест
runTest();
```
