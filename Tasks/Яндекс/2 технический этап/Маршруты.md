// Сортировка билетов
// Дан массив билетов, по которым можно однозначно восстановить маршрут.
// Гарантируется, что входные данные корректны, а каждый город посещался ровно один раз.

// Верните массив с теми же билетами, но отсортированным по пути маршрута.

// Результат
// [
//   { from: "NY", to: "London" },
//   { from: "London", to: "Moscow" },
//   { from: "Moscow", to: "SPb" },
// ];

const tickets = [
    { from: "Moscow", to: "SPb" },
    { from: "NY", to: "London" },
    { from: "London", to: "Moscow" },
];

function sortTickets(tickets) {
    const map = new Map();
    const toCities = new Set();
    const fromCities = new Set();

   // Заполняем структуры данных
    for (let ticket of tickets) {
        map.set(ticket.from, ticket);  // ключ: город, значение объект
        toCities.add(ticket.to);        // запоминаем город прибытия
        fromCities.add(ticket.from);    // запоминаем город отправления
    }
    
	//(откуда вылетели, но куда не прилетали)
    let startCity;

    for (let city of fromCities) {
        if (!toCities.has(city)) {
            startCity = city;
            break;
        }
    }


    const result = [];
    let currentCity = startCity;

    // Пока есть билет из текущего города - двигаемся дальше
    while (map.has(currentCity)) {
        const ticketCurrentCity = map.get(currentCity);
        result.push(ticketCurrentCity);
        currentCity = ticketCurrentCity.to;
    }

    return result;
}
