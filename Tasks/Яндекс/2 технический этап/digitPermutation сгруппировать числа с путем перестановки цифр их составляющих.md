/*
Дан массив целых неотрицательных чисел, нужно сгруппировать друг с другом числа,
которые можно получить путём перестановки цифр их составляющих,
нули при этом игнорируем, т. к. нет числа 011.
Решение должно быть максимально эффективно по памяти и времени.
*/

function digitPermutation(arr) {
    const map = {}

    arr.forEach(digit => {
        const arrDigit = digit.toString()              // "1230"
		 .split('')                // ["1", "2", "3", "0"]
	     .filter(el => el !== '0') // ["1", "2", "3"] (убрали 0)
	     .sort()                  // ["1", "2", "3"] (уже отсортировано)

        if (map[arrDigit]) { // Ищем ключ по отсортированному
            map[arrDigit].push(digit) // Пушим оригинальные
        } else {
            map[arrDigit] = [digit] // По умолчанию
        }
    })

    return Object.values(map)
}
// 
console.log('start test')
console.log(digitPermutation([1230, 99, 23001, 123, 111, 300021, 101010, 90000009, 9]));
// [[99, 90000009], [111, 101010], [1230, 23001, 123, 300021], [9]]
console.log(digitPermutation([11, 22])); // [[11], [22]]
console.log(digitPermutation([11111111112, 12222222222])); // [[11111111112], [12222222222]]
console.log('end test')

function digitPermutation(arr) {
    const groups = new Map();

    // Проходим по каждому числу в массиве
    for (const num of arr) {
        // Создаём массив счётчиков для цифр от 0 до 9
        // Нам нужно знать, сколько раз встречается каждая цифра
        // Например, для числа 112: цифра 1 встречается 2 раза, цифра 2 - 1 раз
        const counts = Array(10).fill(0);
        
        // Работаем с копией числа, чтобы не изменять оригинал
        let n = num;
        
        // Извлекаем цифры числа по одной
        // Пока число больше 0, у него есть цифры
        while (n > 0) {
            // Получаем последнюю цифру (остаток от деления на 10)
            // Например, 123 % 10 = 3
            const digit = n % 10;
            
            if (digit !== 0) {
                // Увеличиваем счётчик для этой цифры
                // counts[3]++ означает, что цифра 3 встретилась ещё раз
                counts[digit]++;
            }
            
            // Отбрасываем последнюю цифру (целочисленное деление на 10)
            // Например, Math.floor(123 / 10) = 12
            n = Math.floor(n / 10);
        }

        // Создаём ключ для группировки
        // counts.slice(1) отбрасывает индекс 0 (цифру 0, которую мы игнорируем)
        // join('') превращает массив в строку
        // Например, [0,2,1,0,...] → "210000000"
        // Этот ключ уникально идентифицирует набор цифр: цифра 1 встретилась 2 раза, цифра 2 - 1 раз
        const key = counts.slice(1).join('') || '0';
        // || '0' нужен для числа 0: когда все цифры нули, ключ будет пустой строкой

        // Группируем числа с одинаковым ключом
        // Если для этого ключа ещё нет массива - создаём
        if (!groups.has(key)) {
            groups.set(key, []);
        }
        // Добавляем текущее число в соответствующую группу
        groups.get(key).push(num);
    }

    // Возвращаем только значения Map (массивы сгруппированных чисел)
    // Нам не нужны ключи, только сами группы
    return Array.from(groups.values());
}
