```js
/**
 * Дан массив точек с целочисленными координатами (x, y).
 * Определить, существует ли вертикальная прямая,
 * делящая точки на 2 симметричных относительно этой прямой набора.
 * Note: Для удобства точку можно представлять не как массив [x, y], а как объект {x, y}
 */
 
 /*
 [0, 0], [0, 0], [1, 1], [2, 2], [3, 1], [4, 0], [4, 0] true
 
	               x
	             x | x
	 ----------xx--|--xx---->
	 Здесь обратить внимание, что нужно именно быть по 2 точки вначале и       в конце
	 
	 Мы можем сделать проекцию только по оси X, найдя серидину 
                  |
	 ----------xxxxxxx---->
*/	 
 
 function isVertSym(...points) {
    if (points.length <= 1) return true;
    
    // Находим среднюю линию
    const xs = points.map(p => p[0]); // Только первые элементы
    const minX = Math.min(...xs); // Min среди первых точек
    const maxX = Math.max(...xs); // Max среди первых точек
    const midX = (minX + maxX) / 2; // Находим середину плюс, так как могут быть такие варианты точек -1 0 1 и середина будет 0, так как мы будем все складывать
    
    // Создаём множество точек для быстрого поиска
    // Нужно учитывать и Y, так как может быть перекос по оси Y, тогда иначе будет неверно
     const pointCount = {};
    
    // Подсчитываем количество каждой точки
    for (let i = 0; i < points.length; i++) {
        const [x, y] = points[i]; // [x, y] = [0, 1]
        const key = `${x},${y}`; // Создаем ключ
        pointCount[key] = (pointCount[key] || 0) + 1; //Если есть ключ, добавляем счетчик, если нет, то устанавливаем 0 
    }
    
    // Проверяем каждую точку
    for (let i = 0; i < points.length; i++) {
        const [x, y] = points[i]; // [x,y] = [0, 4]
        const symmetricX = 2 * midX - x; // Формула, для определения симметричной точки X, относительно мидеаны
        const key = `${symmetricX},${y}`;
		// Ищем точку с тем же y, но с зеркальным X
        // Если симметричной точки нет
        if (!pointCount[key]) {
            return false;
        }
        
        // Уменьшаем счётчик (использовали одну пару)
        // Так как для двух (0,0) нужны две (4,0), но есть только одна → должно быть false, как раз во 2 примере там false
        pointCount[key]--;
    }
    
    return true;
}

```

```js
console.log(isVertSym([0, 0], [0, 0], [1, 1], [2, 2], [3, 1], [4, 0], [4, 0])); // true
console.log(isVertSym([0, 0], [0, 0], [1, 1], [2, 2], [3, 1], [4, 0])); // false
console.log(isVertSym([])); // true
console.log(isVertSym([0, 0])); // true
console.log(isVertSym([0, 0], [10, 0])); // true
console.log(isVertSym([0, 0], [11, 1])); // false
console.log(isVertSym([0, 0], [1, 0], [3, 0])); // false
```
