Нужно написать функцию needleInHaystack. Ответ истинный, если все символы из "иголки" встречаются в "стоге" в том же порядке, в каком они приведены в "иголке", они должны быть слитными.Решение через indexOf / find не допускаются.

```js
function needleInHaystack(needle, haystack) {
    if (needle.length === 0) return true;
    if (needle.length > haystack.length) return false;
    
    // Ищем первый символ иголки
    for (let i = 0; i <= haystack.length - needle.length; i++) {
        if (haystack[i] === needle[0]) {
            let match = true;
            // Проверяем остальные символы
            for (let j = 1; j < needle.length; j++) {
                if (haystack[i + j] !== needle[j]) {
                    match = false;
                    break;
                }
            }
            if (match) return true;
        }
    }
    
    return false;
}

console.log(needleInHaystack('whe', 'cartwheel')); // true (обычное вхождение)

console.log(needleInHaystack('crt', 'cartwheel')); // false (не слитно: c-?-?-t)

console.log(needleInHaystack('cw', 'cartwheel')); // false (c и w есть, но не слитно)

console.log(needleInHaystack('weee', 'cartwheel')); // false (мало e в слове)

  

// Тесты с началом и концом строки

console.log(needleInHaystack('cart', 'cartwheel')); // true (в начале)

console.log(needleInHaystack('heel', 'cartwheel')); // true (в конце)

console.log(needleInHaystack('cartwheel', 'cartwheel')); // true (полное совпадение)

  

// Тесты с повторяющимися символами

console.log(needleInHaystack('aa', 'caartwheel')); // true (две a подряд)

console.log(needleInHaystack('aaa', 'caartwheel')); // false (нужно три a подряд, а их две)

console.log(needleInHaystack('ee', 'cartwheel')); // true (две e подряд есть)

console.log(needleInHaystack('eee', 'cartwheel')); // false (нужно три e)
```
