Дан список целых чисел, повторяющихся элементов в списке нет. Нужно преобразовать это множество в строку, сворачивая соседние по числовому ряду числа в диапазоны. Задача требует преобразовать список уникальных целых чисел в строку, где последовательные числа объединяются в диапазоны через дефис, а одиночные числа выводятся как есть, разделяя группы запятыми; важно корректно обрабатывать как одиночные элементы, так и диапазоны.

```js

function compress(digits) {
    if (!digits.length) return '';

    const sortedArray = [...digits].sort((a, b) => a - b);
    const result = [];
    let start = sortedArray[0];
    let prev = sortedArray[0];

    for (let i = 1; i <= sortedArray.length; i++) {
        const current = sortedArray[i];
        
        // Если текущее число не является следующим по порядку или достигли конца
        if (current !== prev + 1) {
            // Завершаем текущий диапазон
            if (start === prev) {
                result.push(start.toString());
            } else {
                result.push(start + '-' + prev);
            }
            
            // Начинаем новый диапазон
            start = current;
        }
        
        prev = current;
    }

    return result.join(',');
}

// Тесты
console.log(compress([1, 4, 5, 2, 3, 9, 8, 11, 0])); // '0-5,8-9,11'
console.log(compress([1, 4, 3, 2])); // '1-4'
console.log(compress([1, 4])); // '1,4'
console.log(compress([1])); // '1'
console.log(compress([])); // ''

console.log(compress([1, 4, 5, 2, 3, 9, 8, 11, 0])) // '0-5,8-9,11'
console.log(compress([1, 4, 3, 2])) // '1-4'
console.log(compress([1, 4])) // '1,4'
```

Через reduce

```js
function compress(digits) {
    if (!digits.length) return '';

    const sorted = [...digits].sort((a, b) => a - b);
    
    // Используем reduce для построения массива диапазонов
    const ranges = sorted.reduce((acc, current, index, array) => {
        // Если это первый элемент
        if (index === 0) {
            acc.push({ start: current, end: current });
            return acc;
        }
        
        const lastRange = acc[acc.length - 1];
        
        // Если текущее число продолжает последний диапазон
        if (current === lastRange.end + 1) {
            lastRange.end = current;
        } else {
            // Начинаем новый диапазон
            acc.push({ start: current, end: current });
        }
        
        return acc;
    }, []);
    
    // Преобразуем диапазоны в строки
    return ranges.map(range => 
        range.start === range.end 
            ? range.start.toString() 
            : range.start + '-' + range.end
    ).join(',');
}

// Тесты
console.log(compress([1, 4, 5, 2, 3, 9, 8, 11, 0])); // '0-5,8-9,11'
console.log(compress([1, 4, 3, 2])); // '1-4'
console.log(compress([1, 4])); // '1,4'
console.log(compress([1])); // '1'
console.log(compress([])); // ''
```
