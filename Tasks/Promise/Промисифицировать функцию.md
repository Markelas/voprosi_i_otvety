```js
 function fetchData(url, callback) { // задание
    setTimeout(() => {
        callback(`Data from ${url}`);
    }, 2000);
}

function fetchDataPromise(url) { // Моя функция
    return new Promise((resolve) => {
        fetchData(url, resolve)
    })
}

fetchDataPromise('https://api.example.com').then(console.log);
// Промисифицируйте функцию fetchData так, чтобы она возвращала промис.
```
